实现一个简单的基于koa路由和mysql数据库的增删改查demo

安装依赖 ​

json
{
  "dependencies": {
    "koa": "^2.15.3",
    "koa-bodyparser": "^4.4.1", // body解析中间件
    "koa-router": "^12.0.1", // 路由
    "koa2-cors": "^2.0.6", // 解决跨域
    "mysql2": "^3.9.7" // 数据库
  }
}

新建 app.js(或其他名字) ​

项目根目录下新建 app.js 文件

js
// app.js
//引入Koa
const koa= require('koa')
const Router = require('koa-router')

const app = new koa()
const router = new Router()

// 开启这块代码，默认对所有路由生效
// 配置中间件
// app.use( async (ctx)=>{
//     ctx.body = 'hello koa2'
// })

// 仅路由定位到 news 页面生效
router.get('/news',async (ctx)=>{
    ctx.body="this is a demo page"
})

/*启动路由*/
app.use(router.routes())
    .use(router.allowedMethods())

//监听端口
app.listen(3100)

终端运行, 访问 localhost:3100/news，可以看到生效页面

shell
node app.js

作为服务端与数据库使用 ​

  1. 安装数据库 mysql
  • 数据库选择MySQL，官网安装包社区开源版本下载安装
  • 安装完成需要配置环境变量，（如果已关联，跳过）
  • 打开终端CMD，输入账号密码 mysql -u root -p
登录成功后，看到MySQL的命令行提示符，通常是这样的： mysql>
  1. 创建数据库，创建一个名为 my_database 的数据库
shell
CREATE DATABASE my_database;

为了确认数据库是否创建成功，可以使用以下命令来查看所有数据库列表

shell
SHOW DATABASES;

增删改查 ​

  1. 根目录下新建一个文件 mysql.js
js
// mysql.js
const Router = require('koa-router')
const mysql = require('mysql2/promise')

const router = new Router()

const pool = mysql.createPool({
    host: 'localhost',
    user: 'root',
    password: '123456',
    database: 'my_database',
})

// 查询所有用户
router.get('/users', async (ctx) => {
    try {
        const [rows] = await pool.query('SELECT * FROM users')
        ctx.body = rows
    } catch (error) {
        ctx.status = 500;
        ctx.body = { error: 'Error fetching users.' }
    }
})

// 创建新用户
router.post('/users', async (ctx) => {
    try {
        console.log(ctx,'ctx')
        const { name, email } = ctx.request.body
        const [result] = await pool.execute('INSERT INTO users (name, email) VALUES (?, ?)', [name, email])
        ctx.status = 201
        ctx.body = { id: result.insertId, message: 'User added successfully.' }
    } catch (error) {
        console.log(error)
        ctx.status = 500
        ctx.body = { error: 'Error adding user.' }
    }
})

// 更新用户信息
router.put('/users/:id', async (ctx) => {
    const userId = ctx.params.id
    const { name, email } = ctx.request.body
    try {
        await pool.execute('UPDATE users SET name = ?, email = ? WHERE id = ?', [name, email, userId])
        ctx.body = { message: 'User updated successfully.' }
    } catch (error) {
        ctx.status = 500
        ctx.body = { error: 'Error updating user.' }
    }
})

// 删除用户
router.delete('/users/:id', async (ctx) => {
    const userId = ctx.params.id
    try {
        await pool.execute('DELETE FROM users WHERE id = ?', [userId])
        ctx.body = { message: 'User deleted successfully.' }
    } catch (error) {
        ctx.status = 500
        ctx.body = { error: 'Error deleting user.' }
    }
})

module.exports = { newRouter: router }
  1. app.js中引入mysql.js，并更新路由
js
// app.js
const koa= require('koa')
const Router = require('koa-router')
const cors = require('koa2-cors')
const bodyParser = require('koa-bodyparser')

const { newRouter } = require('./mysql')

const app=new koa()
const router = new Router()

app.use(cors())
    .use(bodyParser())
    .use(newRouter.routes())
    .use(newRouter.allowedMethods())

app.listen(3100)
  1. 重新执行 node app.js
  2. 本地新建任意 html ，调用增删改查接口
html

<button id="removeBtn">remove username</button>
<button id="queryBtn">query username</button>
js

const queryBtn = document.querySelector('#queryBtn')
const removeBtn = document.querySelector('#removeBtn')
const updateBtn = document.querySelector('#updateBtn')

  const data = {
      name: 'John2',

  }
  fetch('http://localhost:3100/users', { method: 'post', body: JSON.stringify(data) ,
      headers: {
          'Content-Type': 'application/json'
      }}).then((res, rep) => {
      return res.json()
  }).then(res => {
      console.log(res)
  })
})

      return response.json()

})

})
