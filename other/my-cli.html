const chalk = require("chalk")

function post(url, props, headers = {}, opt = {}) {

    const { noPase = false } = opt
   
    const request = fetch(url, {
        method: "POST",
        headers: {
            Connection: "close",
            "Content-Type": "application/json",
            Accept: "*/*",
            ...headers,
        },
        body: JSON.stringify(props || {}),
    });

    if (noPase) {
        return request
    }

    return request
        .then((data) => data.json())
        .then((res) => {
            const { code, msg } = res
            if (code === 0) {
                return res
            }
            process.exitCode = 99
            process.exit(msg)
            // return Promise.reject(new Error(msg))
        })
}

function get(url, props, headers = {}, opt = {}) {
    const { noPase = false } = opt

    let formatUrl = url
    if (props && Object.keys(props).length > 0) {
        Object.keys(props).forEach((key, index) => {
            const value = props[key]
            formatUrl += `${index === 0 ? "?" : "&"}${key}=${value}`
        })
    }

    const request = fetch(formatUrl, {
        method: "GET",
        headers: {
            Connection: "close",
            "Content-Type": "application/json",
            Accept: "*/*",
            ...headers,
        },
    });

    if (noPase) {
        return request
    }

    return request
        .then((data) => data.json())
        .then((res) => {
            const { code, msg } = res
            if (code === 0) {
                return res
            }

            process.exitCode = 99
            process.exit(msg)
        })
}

// 检查能否访问 npmjs.org 
const checkoutNetwork = () => {
    return get(
        "https://registry.npmjs.org/",

        console.log(chalk.bold.red("访问 npm 失败，请检测网络环境"))
        process.exitCode = 99
        process.exit()
    })
}

检查脚手架版本

js
const fetch = require("node-fetch")
const parse = require("url-parse")
const chalk = require("chalk")
const pkg = require("../../package.json") // package.json 文件

function getCliInfo() {
    const url = "https://registry.npmjs.org/antpkg-cli" // 脚手架发布地址

    return fetch(parse(url, true))
        .then((data) => data.json())
        .then((res) => {
            return res
        })
}
// 检查脚手架版本号
let latestVersion
function versionCheck() {
    return getCliInfo()
        .then((res = {}) => {
            const distTags = res["dist-tags"] || {}
            const { latest } = distTags
            latestVersion = latest
            return pkg.version === latest
        })
        .then((res) => {
            if (!res) {
                // 当前版本不是最新的，
                console.log(chalk.bold.red("当前版本过低，使用如下命令更新："))
                console.log(
                    chalk.bold.green(
                        `  npm install -g antpkg-cli@${latestVersion} --registry https://registry.npmjs.org/`
                    )
                )

                process.exitCode = 99
                process.exit()
            }
        })
}

const fs = require('fs')
const path = require('path')

// 创建新文件
function writeAnt4(copyFile, targetFile) {
   // 测试指定的文件或目录的权限
   fs.access(targetFile, fs.constants.F_OK, (err) => {
      if(err) {
         console.log(`${targetFile} 'does not exist. creat it now'`)

         // 目标目录路径
         const targPathList = targetFile.split('/')

         if(targPathList.length === 1) {
            // 将源文件内容写入目标文件(这里的文件是通过导出变量名 allCopy对象中 本地进行维护。)
            // 也可以使用 fs.readFile(sourceFilePath, (err, data) => {})异步读取文件的全部内容，再读取data数据写入文件
            fs.writeFile(targetFile, allCopy[copyFile], (err) => {
               if (err) {
                  console.error(err)
                  return
               }

               console.log(`${targetFile} 文件拷贝成功`)
            })
            return
         }

         const targetDirectory =  targPathList.slice(0, targPathList.length - 1).join('/')

         // 确保目标目录存在
         fs.mkdir(targetDirectory, { recursive: true }, (err) => {
            if (err) {
               console.error(`${err} 创建目录中`)
               fs.mkdirSync(targetDirectory)
            }

            // 将源文件内容写入目标文件
            fs.writeFile(targetFile, allCopy[copyFile], (err) => {
               if (err) {
                  console.error(err)
                  return
               }

               console.log(`${targetFile} 文件拷贝成功`)
            })
         })
      }else {
         console.log(`${targetFile} 'exists'`)
      }
   })
}

// 已有文件中插入或替换内容
function reWriteAnt4(copyFile, targetFile, indexStr, replaceIt = false, insertBefore) {
   fs.access(targetFile, fs.constants.F_OK, (err) => {
      if(err) {
         console.log(`${targetFile} 'does not exist. please creat it`)
      }else {
         fs.readFile(targetFile, (err, data) => {
            if (err) {
               console.log('读取文件时发生错误：', err)
            } else {
               // 找到要插入代码的位置
               const insertPositionIndex = data.indexOf(indexStr)

               if (insertPositionIndex !== -1) {
                  // 防止重复写入
                  if(!replaceIt && data.indexOf(allCopy[copyFile].replaceAll(/[\n\t]/g, '')) !== -1) {
                     return
                  }
                  // 在目标位置插入代码
                  const updatedData = replaceIt ? allCopy[copyFile] : data.slice(0, insertPositionIndex + indexStr.length) + allCopy[copyFile] + data.slice(insertPositionIndex + indexStr.length)

                  // 写入修改后的内容
                  fs.writeFile(targetFile, updatedData, (err) => {
                     if (err) {
                        console.log('写入文件时发生错误：', err)
                     } else {
                        console.log(`代码已成功插入到 ${targetFile} 文件中！`)
                     }
                  })
               } else {
                  console.log('找不到插入位置！尝试写入')
                  if(insertBefore) {
                     const insertPositionIndex = data.indexOf(insertBefore)
                     const updatedData = data.slice(0, insertPositionIndex) + allCopy[copyFile] + data.slice(insertPositionIndex)
                     fs.writeFile(targetFile, updatedData, (err) => {
                        if (err) {
                           console.log('写入文件时发生错误：', err)
                        } else {
                           console.log(`代码已成功插入到 ${targetFile} 文件中！`)
                        }
                     })
                  }else {
                     console.log('无写入位置，已取消写入')
                  }
               }
            }
         })
      }
   })
}

  1. bin.js 执行命令交互函数
js
const inquirer = require('inquirer')
const { Command } = require('commander')

const program = new Command()

checkoutNetwork().then(async _ => {
   await nodeVersionCheck()
   await versionCheck()
   main()
})

function main () {
   program
      .command('init')
      .action((str, options) => {
          inquirer.prompt(promptTypeList).then(res => {
           const { val, installCommand, copyFiles, reWriteFiles = [], insertBefore = '' } = res.type
           if(val === 'pkg-ant4') {
              startLoading()
              executeInstall(installCommand).then(_ => {
                 stopLoading('依赖下载成功，替换文件中')
                 copyFiles.forEach(item => {
                    writeAnt4(item.copyFile, item.targetFile)
                 })
                 reWriteFiles.forEach(item => {
                    reWriteAnt4(item.copyFile, item.targetFile, item.indexStr, item.replaceIt, item.insertBefore)
                 })
              }).catch(_ => {
                 failLoading('下载依赖失败，请检查或重试')
              })
           }
        })
      })

   program.parse()
}
