  <div class="dot-wrap"></div>
</div>
css
.loading-wrap {
      width: 100%;
      height: 100%;
      position: fixed;
      left: 0;
      top: 0;
      z-index: 1000;
      background-color: #fff;
      opacity: 0.5;
      display: flex;
      flex-direction: row;
      align-items: center;
      justify-content: center;
}

  .dot-wrap {
      width: 30px;
      height: 20px;
      position: relative;
  }

  .dot-wrap:before,
  .dot-wrap:after {
      content: "";
      display: block;
      width: 8px;
      height: 8px;
      border-radius: 50%;
      background: #00bebe;
      position: absolute;
  }

  .dot-wrap:before {
      transform: translateX(0);
      animation: loading-left 1.2s infinite;
  }

  .dot-wrap:after {
      background: lighten(#00bebe, 8%);
      transform: translateX(22px);
      animation: loading-right 1.2s infinite;
  }

  @keyframes loading-left {
      from {
          transform: translateX(0);
      }

      50% {
          transform: translateX(22px);
      }

      to {
          transform: translateX(0);
      }
  }

  @keyframes loading-right {
      from {
          transform: translateX(22px);
      }

      50% {
          transform: translateX(0);
      }

      to {
          transform: translateX(22px);
      }
  }

css
* {
    box-sizing: border-box;
}

body {
    display: grid;
    place-items: center;
    min-height: 100vh;
    overflow: hidden;
}

.loader {
    --size: 25vmin;
    height: var(--size);
    width: var(--size);
    position: relative;
    animation: spin 0.75s infinite steps(var(--count));
}

.loader span {
    position: absolute;
    height: 25%;
    width: 10%;
    background: grey;
    top: 50%;
    left: 50%;
    transform: translate(-50%, -50%)
    rotate(calc(((360 / var(--count)) * var(--index)) * 1deg))
    translate(0, -125%);
    opacity: calc(var(--index) / var(--count));
}

@keyframes spin {
    to {
        transform: rotate(360deg);
    }
}
