一些注意点 ​

渲染svg内容的方式 ​

渲染svg常见得两种情况：

  1. html中直接嵌入
html
<body>
    <svg xmlns="http://www.w3.org/2000/svg" id="svg">
      <circle cx="100" cy="100" r="50" fill="blue" id="circle" />
    </svg>
</body>
  1. 远程获取svg文件并解析放入目标容器
js
fetch('path/file.svg')
    .then(response => response.text())
    .then(svgContent => {
        const parser = new DOMParser()
        const svgDocument = parser.parseFromString(svgContent, 'image/svg+xml')
        const svgElement = svgDocument.documentElement

        // Append the SVG to the target
        svgContainer.appendChild(svgElement)
    })

本次记录使用方式2 远程获取本地svg

关于svg文件命名空间导致的问题 ​

  1. xmlns: 这是默认命名空间声明，浏览器和 SVG 引擎通常对其处理最为直接和兼容。
  2. xmlns:xlink: 这是用于定义 XLink 命名空间，专门用于处理链接。
  3. xmlns:cge: 这是一个自定义命名空间，不属于 SVG 标准。

INFO

SVG 元素置于 xmlns:xlink、xmlns:cge 命名空间下会导致浏览器无法正确识别这些元素，从而导致拖拽和平移等事件无法正常工作（实际使用中，未缩放状态下即初始状态下平移 svg无法与鼠标移动同步）。

因此需要手动删除这些xmlns:xlink、xmlns:cge 命名空间，对应远程获取的svg文件，可在解析后，通过移除属性的方式删除

js
// Remove the incorrect namespaces
 svgElement.removeAttribute("xmlns:xlink")
 svgElement.removeAttribute("xmlns:cge")

将解析的svg文件内容全部放入放入一个新的 <g> 标签，并将 SVG 插入到目标元素中 ​

将 SVG 元素内容放入一个新的 <g> 标签，并将 SVG 插入到目标元素中，主要出于以下考虑：

  1. 使用 <g> 标签可以更好地组织和管理 SVG 元素，便于应用整体变换（缩放和平移）而不影响原始 SVG 的定义
  2. 直接放入可能导致 SVG 内部定义的 ID 或其他全局属性与外部冲突, 使用 <g> 标签可以将 SVG 的内容隔离在一个组中，减少这种冲突的风险

js
const svgContainer = document.getElementById('svg-container')

// Fetch and render SVG
fetch('./file.svg')
    .then(response => response.text())
    .then(svgContent => {
        const parser = new DOMParser()
        const svgDocument = parser.parseFromString(svgContent, 'image/svg+xml')
        const svgElement = svgDocument.documentElement

        // Remove the incorrect namespaces
        svgElement.removeAttribute("xmlns:xlink")
        svgElement.removeAttribute("xmlns:cge")

        const gElement = document.createElementNS('http://www.w3.org/2000/svg', 'g')

        // Move all child nodes from the SVG to the <g> element
        while (svgElement.firstChild) {
            gElement.appendChild(svgElement.firstChild)
        }

        // Append the <g> element back to the SVG
        svgElement.appendChild(gElement)

        svgContainer.appendChild(svgElement)

        // save mouse position
        let lastPosition = {}, position = {}
       // Drag to pan
       let isDragging = false, initialMatrix

       // return position object
        function getPoint(event) {
            return {
                x: event.clientX,
                y: event.clientY
            }
        }
        
        svgContainer.addEventListener('mousedown', (event) => {
            isDragging = true
            lastPosition = getPoint(event)
            const transform = gElement.transform.baseVal.consolidate()
            initialMatrix = transform ? transform.matrix : svgElement.createSVGMatrix()
        })

        document.addEventListener('mousemove', (event) => {
            if (isDragging) {
                position = getPoint(event)
                // calc translation with scale
                const delta = {
                    dx: (position.x - lastPosition.x) / initialMatrix.a,
                    dy: (position.y - lastPosition.y) / initialMatrix.d
                }
                
                const translationMatrix = svgElement.createSVGMatrix().translate(delta.dx, delta.dy)
                const matrix = initialMatrix.multiply(translationMatrix)
                const mstr = `matrix(${matrix.a}, ${matrix.b}, ${matrix.c}, ${matrix.d}, ${matrix.e}, ${matrix.f})`
                gElement.setAttribute('transform', mstr)

                // Update the transform attribute with the new matrix
                // gElement.transform.baseVal.initialize(svgElement.createSVGTransformFromMatrix(matrix))
            }
        })

        document.addEventListener('mouseup', () => {
            isDragging = false
            lastPosition = position
            svgElement.style.cursor = 'grab'
        })

        svgElement.addEventListener('wheel', (event) => {
            event.preventDefault()

            const scaleAmount = event.deltaY > 0 ? 0.9 : 1.1

            // Get the current mouse position relative to the SVG
            const pt = svgElement.createSVGPoint()
            pt.x = event.clientX
            pt.y = event.clientY
            const cursorPt = pt.matrixTransform(svgElement.getScreenCTM().inverse())

            const transform = gElement.transform.baseVal.consolidate()
            const matrix = transform ? transform.matrix : svgElement.createSVGMatrix()

            // Create the scale matrix at the cursor point
            const scaleMatrix = svgElement.createSVGMatrix()
                .translate(cursorPt.x, cursorPt.y)
                .scale(scaleAmount)
                .translate(-cursorPt.x, -cursorPt.y)

            // Apply the scale matrix to the current transform matrix
            const newMatrix = matrix.multiply(scaleMatrix)
           
            // ① Set the value matrix of the transformer for the attribute
            const mstr = `matrix(${newMatrix.a}, ${newMatrix.b}, ${newMatrix.c}, ${newMatrix.d}, ${newMatrix.e}, ${newMatrix.f})`
            gElement.setAttribute('transform', mstr)
           
            // ② Update the transform attribute with the new matrix，use baseVal.initialize
            // gElement.transform.baseVal.initialize(svgElement.createSVGTransformFromMatrix(newMatrix))
        })

        // Disable image dragging for better UX
        svgElement.addEventListener('dragstart', (event) => event.preventDefault())
    })
    .catch(error => console.error('Error fetching SVG:', error))

TIP

可以看到高亮的 <use> xlink:href 引用的是 <symbol> 的 id，这个时候直接对<use>设置任何 fill和stroke的值都是不生效的，而如果直接对引用的 <symbol>设置 fill和stroke会导致所有相同引用的图元都变化，无法实现只更改当前选中的 <use> 元素所引用的 <symbol> 内部元素的高亮样式。

这个时候用了一个折中的办法，拷贝一份当前引用的 <symbol>，对克隆的 <symbol> 更改样式，并将克隆 <symbol> 放入 <defs>中，并且改变 <use>指向该克隆的指引。demo代码实现如下 ​

js
const svgEle = document.querySelector('#svg')
  const  panZoomTiger = svgPanZoom(svgEle, {
      maxZoom: 20,
  })

  // 用于存储已复制的 <symbol> 元素，避免重复复制
  const gMaps = new Map()
  let lastHighlightedId = ''

  svgEle.addEventListener('click', event => {
      const node = event.target.parentNode
      if(lastHighlightedId === node.id) return
      removeLastHighlight()
      // 这里条件分支是区分实际情况，`<g>` 图元下种类不同的维护，有的是 `<use>`,有的使用path路径
      if(event.target.tagName === 'use' && node) {
          if(node.id) {
              lastHighlightedId = node.id
              const gEle = document.querySelector(`#${node.id}`)
              gEle.classList.add('highlighted')

              // gEle.querySelector('use').classList.add('highlighted')
              // xlink:href 引用链接含有特殊字符 @ 需要转义
              const symbol = document.querySelector(gEle.querySelector('use').getAttribute('xlink:href').replaceAll('@', '\\@'))

              gMaps.set(node.id, {
                  originalSymbolId: symbol.getAttribute('id'),
                  newSymbolId: '',
                  clonedSymbolElSets: new Set, 
                  loopEnd: false // 递归结束标志
              })
              // 复制并修改样式的函数。
              function cloneAndModifySymbol(originalSymbol) {
                  // 检查是否已经复制过这个 <symbol>
                  if(gMaps.has(node.id) && gMaps.has(node.id).loopEnd) {
                      const arr = Array.from(gMaps.get(node.id).clonedSymbolElSets)
                      return arr[arr.length - 1]
                  }

                  // 创建一个副本，并修改样式
                  const clonedSymbol = originalSymbol.cloneNode(true)

                  // 修改副本的样式，此处是对svg中不同图元的维护
                  const circles = clonedSymbol.querySelectorAll('circle')
                  const lines = clonedSymbol.querySelectorAll('line')
                  const rects = clonedSymbol.querySelectorAll('rect')
                  const polygons = clonedSymbol.querySelectorAll('polygon')
                  const polylines = clonedSymbol.querySelectorAll('polyline')

                  const pathList = [circles, lines, rects, polygons, polylines]
                  for (const elements of pathList) {
                      elements.forEach(item => {
                          if(item.getAttribute('fill') !== 'none') {
                              item.setAttribute('fill', 'yellow');
                          }
                          if(item.getAttribute('stroke') !== 'none') {
                              item.setAttribute('stroke', 'yellow');
                          }
                      })
                  }

                  // 递归处理内部的 <use> 元素引用
                  const innerUses = clonedSymbol.querySelectorAll('use')
                  innerUses.forEach(function(innerUse) {
                      const innerHref = innerUse.getAttribute('xlink:href')
                      const innerSymbol = document.querySelector(innerHref)
                      const clonedInnerSymbol = cloneAndModifySymbol(innerSymbol)
                      innerUse.setAttribute('xlink:href', '#' + clonedInnerSymbol.getAttribute('id'))
                  })

                  // 给副本生成一个新的 id，并添加到 <defs> 中
                  const newSymbolId = 'highlightedSymbol_' + Math.random().toString(36).substr(2, 9)
                  clonedSymbol.setAttribute('id', newSymbolId)
                  const defs = document.querySelector('defs')
                  defs.appendChild(clonedSymbol)

                  // 记录已经复制过的 <symbol>
                  // 保存递归过程中的每一个新生成的 Symbol 元素，为下一次清除使用
                  gMaps.get(node.id).clonedSymbolElSets.add(clonedSymbol)
                  // 保存递归结束后的 symbol id，即被高亮 <use> 最外层 symbol href的引用
                  gMaps.get(node.id).newSymbolId = newSymbolId

                  return clonedSymbol
              }

              // 创建并应用修改后的 <symbol>
              const modifiedSymbol = cloneAndModifySymbol(symbol)
              gMaps.get(node.id).loopEnd = true
              gEle.querySelector('use').setAttribute('xlink:href', '#' + modifiedSymbol.getAttribute('id'))
          }
      }
      else if((event.target.tagName === 'polyline' || event.target.tagName === 'text') && node) {
          lastHighlightedId = node.id
          const gEle = document.querySelector(`#${node.id}`)
          gEle.classList.add(...['highlighted', 'colored'])
      }

      function removeLastHighlight() {
          if(lastHighlightedId) {
              const gEle = document.querySelector(`#${lastHighlightedId}`)
              gEle.classList.remove(...['highlighted', 'colored'])
              // 指向更改前的href, 并删除上一次创建的所有symbol
              if(gMaps.get(lastHighlightedId)) {
                  gEle.querySelector('use').setAttribute('xlink:href', `#${gMaps.get(lastHighlightedId).originalSymbolId}`)
                  for (const symbolElement of gMaps.get(lastHighlightedId).clonedSymbolElSets) {
                      symbolElement.remove()
                  }
              }
              lastHighlightedId = ''
          }
      }
  })

其中需要注意的点：

  1. 每次点击事件需要判断是否为目标高亮图元，不是则清除上一次高亮，这里通过判断当前节点di与上一次高亮id，前置了清除事件
  2. 对于目标图元下纯path路径，直接添加样式变更即可，重点在引用图元
  3. 对于引用 <symbol> 图元，需要递归处理 <symbol>，因其内部也可能嵌套 <use>
