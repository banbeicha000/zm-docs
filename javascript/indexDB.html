let indexedDB = window.indexedDB, db, dbName='hellIndexedDB', version=1, storeName='helleStore'

  request.onsuccess=function (event) {

      let objectStore
      if(!db.objectStoreNames.contains(objectStore)) {
          objectStore=db.createObjectStore(storeName,{keyPath:'id'})
          // objectStore.createIndex('name', 'name', { unique: true }) // 创建索引 可以让你搜索任意字段
      }
  }

  // 添加数据
  function addData(db, storeName, data) {
      let request = db.transaction([storeName], 'readwrite') // 事务对象 指定表格名称和操作模式（"只读"或"读写"）
          .objectStore(storeName) // 仓库对象
          .add(data)

      request.onsuccess = function (event) {
          console.log('数据写入成功')
      }

      request.onerror = function (event) {
          console.log('数据写入失败')
          throw new Error(event.target.error)
      }
  }
  // 根据id获取数据
  function getDataByKey(db, storeName, key) {
      let transaction = db.transaction([storeName]) // 事务
      let objectStore = transaction.objectStore(storeName) // 仓库对象
      let request = objectStore.get(key)

      request.onerror = function(event) {
          console.log('事务失败')
      }

      request.onsuccess = function(event) {
          console.log('主键查询结果: ', request.result)
      }
  }

  // 根据id修改数
  function updateDB(db, storeName, data) {
      let request = db.transaction([storeName], 'readwrite') // 事务对象
          .objectStore(storeName) // 仓库对象
          .put(data)

      request.onsuccess = function() {
          console.log('数据更新成功')
      }

      request.onerror = function() {
          console.log('数据更新失败')
      }
  }

  // 根据id删除数据
  function deleteDB(db, storeName, id) {
      let request = db.transaction([storeName], 'readwrite').objectStore(storeName).delete(id)

      request.onsuccess = function() {
          console.log('数据删除成功')
      }

      request.onerror = function() {
          console.log('数据删除失败')
      }
  }

  // 由于打开indexDB是异步的加个定时器避免 db对象还没获取到值导致 报错

      updateDB(db, storeName, {id: 1638164880484, desc: '修改的内容'})

结合web worker使用 ​

INFO

web worker无法操作主线程中的 IndexedDB 实例，即无法将主线程的 IndexedDB 实例序列化通信给web worker。但是，可以在 Web Worker 中创建和操作一个 IndexedDB 实例，前提是数据库和数据结构在主线程和 Web Worker 中是兼容的。

应用 ​

在vue2项目中使用，使用 IndexedDB 存储 element-ui的 tree组件所需的大量数据 （~10M），web worker从 IndexedDB 取数据。 ​

TIP

其中 new Worker('/work.js') 的 work.js 文件应放在项目的 public 目录下，否则会报错 Failed to load module script: The server responded with a non-JavaScript MIME type of "text/html". Strict MIME type checking is enforced for module scripts per HTML spec. ，原因是服务器返回的 MIME 类型是 text/html，而不是正确的 JavaScript MIME 类型 (text/javascript)

js
// 存储
import { regionData } from './data.js' // 接口数据

const dbName = "tree_data"

const request = indexedDB.open(dbName, 1)

request.onerror = (event) => {
    // 错误处理
}

request.onupgradeneeded = (event) => {
    const db = event.target.result

    // 创建一个对象存储来存储信息，将“name”作为键路径
    const objectStore = db.createObjectStore("customers", { keyPath: "name" })

    // 创建一个索引以通过姓名来搜索
    objectStore.createIndex("name", "name", { unique: true })

    // 使用事务的 oncomplete 事件确保在插入数据前对象存储已经创建完毕。
    objectStore.transaction.oncomplete = (event) => {
        // 将数据保存到新创建的对象存储中。
        const customerObjectStore = db
            .transaction("customers", "readwrite")
            .objectStore("customers")

        const saveObj = {
            name: 'regionData',
            obj: regionData
        }

        customerObjectStore.add(saveObj)
    }
}

// work.js
self.onmessage = event => {

    const { dbName, storeName } = event.data

    const request = indexedDB.open(dbName)

    request.onsuccess = function(event) {
        const db = event.target.result
        const transaction = db.transaction([storeName], 'readonly')
        const store = transaction.objectStore(storeName)

        const getRequest = store.get('regionData')
        // 或者使用索引查找。这种指定查找效率更高
        // const index = store.index('name')
        // const getRequest = index.get('regionData')
        
        // 所有与 IndexedDB 相关的操作都是异步完成的，并且只有在操作完成时，才去访问 result 属性
        // 否则会报错 `Uncaught DOMException: Failed to read the 'result' property from 'IDBRequest': The request has not finished`
        getRequest.onsuccess = function(ev) {
            const result = ev.target.result
            if (result) {
                self.postMessage({ data: result.obj })
            } else {
                self.postMessage({ error: 'No data found for the given key.' })
            }
        }
    }
}

// indexedDB.js
const apiURL = 'https://example.com/api/data' // API接口地址
const dbName = 'MyDatabase'
const storeName = 'DataStore'
const dataKey = 'cachedData'

// 创建并打开数据库
function openDatabase() {
    return new Promise((resolve, reject) => {
        const request = indexedDB.open(dbName, 1)

        request.onerror = () => reject('数据库打开失败')

        request.onsuccess = (event) => {
            resolve(event.target.result)
        }

        request.onupgradeneeded = (event) => {
            const db = event.target.result
            if (!db.objectStoreNames.contains(storeName)) {
                db.createObjectStore(storeName, { keyPath: 'id' })
            }
        }
    })
}

// 获取API数据并存储到 IndexedDB 中
async function fetchAndStoreData(db) {
    try {
        const response = await fetch(apiURL)
        const result = await response.json()
        const data = {
            id: dataKey,
            value: result.data,
            timestamp: new Date().toISOString()
        }

        const transaction = db.transaction(storeName, 'readwrite')
        const store = transaction.objectStore(storeName)
        store.put(data)

        console.log('数据已更新并存储到 IndexedDB')
        return data.value
    } catch (error) {
        console.error('获取数据失败:', error)
    }
}

// 检查数据是否失效（根据日期）
function isDataExpired(timestamp) {
    const today = new Date().toDateString()
    const dataDate = new Date(timestamp).toDateString()
    return today !== dataDate // 如果不是同一天，则表示数据过期
}

// 获取数据
async function getData() {
    const db = await openDatabase()
    const transaction = db.transaction(storeName, 'readonly')
    const store = transaction.objectStore(storeName)
    const request = store.get(dataKey)

    return new Promise((resolve, reject) => {
        request.onsuccess = async (event) => {
            const result = event.target.result

            if (result && !isDataExpired(result.timestamp)) {
                console.log('使用缓存数据')
                resolve(result.value)
            } else {
                console.log('数据失效，重新获取')
                const data = await fetchAndStoreData(db)
                resolve(data)
            }
        }

        request.onerror = () => reject('获取数据失败')
    })
}

// 主线程代码
export async function getTree() {
    // 创建 Web Worker 实例
    const worker = new Worker('/worker.js')

    return new Promise((resolve, reject) => {
        // 监听来自 Web Worker 的消息
        worker.onmessage = (event) => {
            console.log(event,'work event')
            const { success, data, error } = event.data
            if(success) {
                console.log('Data received from Web Worker:', data)
                resolve(data)
            }else {
                reject(new Error(error))
            }
        }
        // 向 Web Worker 发送消息
        worker.postMessage({ action: 'getDataById' })
    })
}

然后实际使用中，虽然解决了占用主线程的问题，但对于当前页面需要立即获取并显示的部分，indexedDB 查询并获取到数据会有可感知到的延迟（1~2秒）。通过查找文档，发现这是不可避免的。可能优化的方向有2点

  1. 查找方式不管是 get 方法指定查找 还是创建索引，已经是查找高效了
  2. indexedDB 存储和获取涉及序列化与反序列化，对于复杂数据，本身比 JSON.stringify 更高效，无需处理
  3. 对于树型结构数据，分段存储不可取，受限当前组件所需数据也是树型结构

折中方案是 使用 loading 提示，或者显示一个基础数据结构，当 indexedDB 获取到值后整体替换。 ​

使用ServiceWorker ​

  1. Service Worker 缓存主要适合快速缓存 HTTP 响应，比如接口的 JSON 数据、静态资源等，使用方便且读取响应速度较快
  2. Service Worker 缓存使用 Cache API，专为网络响应设计。Cache API 是键值对形式（请求-响应），数据不需要手动序列化和反序列化，适合存储 HTTP 响应
  3. Cache API 的数据读取速度通常快于 IndexedDB，因为缓存响应已经是浏览器的二进制格式，尤其适合小到中等规模的数据。当请求匹配时，Service Worker 缓存可以直接返回整个响应对象，减少额外的解析操作
  4. Cache API 设计为临时缓存，适合“取一用一”的情况，对于复杂的数据存储和搜索索引不太合适
js
// service-worker.js
const CACHE_NAME = 'api-cache-v1'

// <------- 清理旧缓存
self.addEventListener('install', event => {
    // 跳过等待，直接进入激活状态
    event.waitUntil(self.skipWaiting())
})
self.addEventListener('activate', event => {
    // 清理旧缓存
    event.waitUntil(
        caches.keys().then(cacheNames => {
            return Promise.all(
                cacheNames.filter(cacheName => cacheName !== CACHE_NAME)
                    .map(cacheName => caches.delete(cacheName))
            )
        })
    )
    return self.clients.claim()
})
// ------->

self.addEventListener('fetch', event => {
    const url = event.request.url

    // 只缓存特定 API 请求
    if (url.includes('/api/areas')) {
        event.respondWith(
            caches.open(CACHE_NAME).then(async cache => {
                // 检查缓存是否已有该请求
                const cachedResponse = await cache.match(event.request)
                if (cachedResponse) {
                    return cachedResponse
                }

                // 如果没有缓存，请求网络并缓存响应
                const networkResponse = await fetch(event.request)
                cache.put(event.request, networkResponse.clone())
                return networkResponse
            })
        )
    }
})

TIP

  1. Service Worker 只能在 HTTPS 或 localhost 环境下工作。如果是在 HTTP 环境下，Service Worker 不会触发 fetch 事件
  2. Service Worker 只能拦截同域请求，如果请求的接口是外部接口（如跨域请求），则 Service Worker 无法拦截。如果有条件将数据通过后端代理服务请求，使请求变为同源
  3. 默认情况下，Service Worker 只会在其所在目录及以下的路径生效。如果 service-worker.js 文件位于 /public/service-worker.js，则它的作用域是 /，可以监听所有的网络请求
  4. 定期手动清理资源，关闭页面不会导致 Service Worker 被清理
