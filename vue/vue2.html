element-ui ​

INFO

在vue2的项目中，使用element组件的频率还是比较高的。而在业务中，涉及到列表查询和编辑的情况是普遍的。 对组件进行二次封装可以“优雅”得复用，减少重复开发代码段（拷贝错漏等）的失误，以及更好得维护。因此，基于个人经历的业务，二次封装出较通用的组件

form-row ​

结合el-row布局与常用的输入、下拉等编写可配置的表单组件

vue
<script>
import { cloneDeep, debounce, isArray, isString } from "lodash-es"

const childCompTrans = {
  'el-select': 'el-option',
  'el-radio-group': 'el-radio'
}
export default {
  name: "formRow",
  props: {
    labelPosition: {
      type: String,
      validator(value) {
        return ['right', 'left'].includes(value)
      },
      default: 'right'
    },
    labelWidth: {
      type: String,
      default: '120px'
    },
    formDisabled: {
      type: Boolean,
      default: false
    },
    // 表单form属性
    formAttrs: {
      type: Object,
      default() {
        return {}
      }
    },
    formItems: {
      type: Array,
      default() {
        return []
      }
    },
    forms: {
      type: Object,
      default() {
        return {}
      }
    },
    visible: {
      type: Boolean,
      default: false
    },
    gutter: {
      type: Number,
      default: 20
    },
    labelHiddenLength: {
      type: Number,
      default: 12
    },
    alongContentWidth: {
      type: Boolean,
      default: true
    }
  },
  data() {
    return {
      formModel: this.forms,
      defaultAttrs: {
        clearable: true
      },
      selectOpMinWid: 0
    }
  },
  methods: {
    handleApiForm(item) {
      this.$emit('handleBtn', item)
    },
    handlePlaceHoder({comp, label}) {
      let labelStr = this.dealLabel(label)
      if(comp === 'el-input') {
        return `请输入${labelStr}`
      }else if(comp === 'el-select' || comp === 'el-date-picker') {
        return `请选择${labelStr}`
      }
    },
    dealLabel(label) {
      let labelStr
      if(label.indexOf(':')) {
        let strList = label.split(':')
        labelStr = strList.slice(strList-1, 1).join('')
      }else {
        labelStr = label
      }
      return labelStr
    },
    handleRuleMsg(label) {
      return `请将${this.dealLabel(label)}填写完整`
    },
    handleFocus(e, focusName, { item, formModel }) {
      if(focusName) {
        this.$emit(focusName, { focusName, item, formModel })
      }
      // 下拉框宽度
      if(item.comp === 'el-select') {
        this.selectOpMinWid = e.target.clientWidth
      }
    },
    handleChange(changeName, { item, formModel }) {
      if(changeName) {
        this.$emit(changeName, { changeName, item, formModel })
      }
    },
    // 远程搜索
    remoteMethod(query, item) {
      let use_debound
      if(item.useApiRemote) {
        use_debound = debounce(() => {
          item.loadQuery.pageNo = 1
          if(isArray(item.childOptions)) {
            item.childOptions.splice(0)
          }else if(isArray(item.childOptions.options)) {
            item.childOptions.options.splice(0)
          }
          this.$emit(item.attrs.remoteMethod, { isFirst: false, item, load: false, queryStr: query, ...item.loadQuery })
        }, 200)
      }else if(!item.useApiRemote) {
        use_debound = debounce(() => {
          let opKey = `${item['prop']}Options`
          if(!isArray(item[`${opKey}`])) return
          let serchOption = [], num = item.loadNum || 20
          if(query) {
            serchOption = item[`${opKey}`].filter(it => {
              return isString(it.label) ? it.label.toLowerCase().indexOf(query.toLowerCase()) > -1 : it.label === Number(query)
            })
          }else {
            serchOption = item[`${opKey}`].slice(0, num)
          }
          if(isArray(item.childOptions)) {
            item.childOptions.splice(0)
            item.childOptions.push(...serchOption)
          }else if(isArray(item.childOptions.options)) {
            item.childOptions.options.splice(0)
            item.childOptions.options.push(...serchOption)
          }
        }, 200)
      }
      use_debound()
    },
    justifyValid(cb, otherCb) {
      const p1 = this.$refs.formModel.validate()
      p1.then(_ => {
        if(typeof cb === 'function') {
          try {
            cb()
          }catch (err) {
            console.error(err)
          }
        }
      }).catch(_ => {
        if(typeof otherCb === 'function') {
          try {
            otherCb()
          }catch (err) {
            console.error(err)
          }
        }
      })
    },
    clearValidate() {
      this.$nextTick(() => {
        this.$refs.formModel.clearValidate()
      })
    },
    resetForm() {
      for (const formKey in this.formModel) {
        if(typeof this.formModel[formKey] === 'string') {
          this.formModel[formKey] ? this.formModel[formKey] = '' : ''
        }else if(Array.isArray(this.formModel[formKey]) && this.formModel[formKey].length) {
          this.formModel[formKey].splice(0)
        }else if(typeof this.formModel[formKey] === 'number') {
          this.formModel[formKey] = null
          // this.formModel[formKey] !== 0 ? this.formModel[formKey] = null : ''
        }
      }
    },
    /*
    *  childOptions 有2种数据格式
    *  ① 数组：[ comp: 'childCompName', attrs: {}, label: '' ] （comp 子元素组件名称）
    *  ② 对象：{ options: [{ label: '', value: '' }], attrs: {} } （这种方式通过 childCompTrans 维护子元素组件）
    * */
    dealChildOptions(options, item) {
      if(Array.isArray(options)) return options
      if(typeof options === 'object') {
        if(!Array.isArray(options.options)) return []
        let arr = []
        arr.push(...options.options.map(it => {
          return {
            attrs: { ...it, ...(options['options']['attrs'] ? options['options']['attrs'] : {})},
            comp: childCompTrans[item.comp],
            label: item.comp === 'el-radio-group' ? it['label'] : null
          }
        }))
        return arr
      }
      return []
    },
    // treeSelect组件手动更新并校验
    handleUpdateComp({ compKey, value }) {
      // this.formModel[compKey] = value
      // upload组件未双向绑定，特殊处理
      if(Array.isArray(value)) {
        this.$set(this.formModel, compKey, [])
        this.formModel[compKey] = value
      }else {
        this.formModel[compKey] = value
      }
      if(!value && value !== 0) return
      let findInd = this.$refs.formModel.fields.findIndex(item => item.prop === compKey)
      if(~findInd) {
        this.$nextTick(() => {
          this.$refs.formModel.fields[findInd].clearValidate()
        })
      }
    }
  },
  watch: {
    visible(newVal) {
      if(newVal) {
        this.clearValidate()
      }else {
        this.resetForm()
      }
    }
  },
  mounted() {
    // 初始加载清除校验，主要针对多选情况
    this.clearValidate()
  },
  directives: {
    lazyLoad: {
      inserted(el, binding, vnode, oldVnode) {
        if(binding.value.isLazy && binding.value.item.comp === 'el-select') {
          if(!binding.value.item.childOptions) return
          let onceLoad = binding.value.item.loadNum || 20
          let startNum = 0, endNum = 0, nowCount = 0, loadLen = 0, allOptions = []
          if(!binding.value.item.useApiRemote) {
            const dealOption = (options, item) => {
              if(Array.isArray(options)) return options
              if(typeof options === 'object') {
                if(!Array.isArray(options.options)) return []
                return options['options']
              }
              return []
            }
            allOptions = cloneDeep(dealOption(binding.value.item.childOptions, binding.value.item))
            if(allOptions.length === 0) return
            binding.value.item[`${binding.value.item['prop']}Options`] = allOptions
            loadLen = Math.ceil(allOptions.length / onceLoad)
            startNum = nowCount * onceLoad
            endNum = (nowCount + 1) * onceLoad
            if(Array.isArray(binding.value.item.childOptions)) {
              binding.value.item.childOptions.splice(0)
              binding.value.item.childOptions.push(...allOptions.slice(startNum, endNum))
            }else {
              binding.value.item.childOptions.options.splice(0)
              binding.value.item.childOptions.options.push(...allOptions.slice(startNum, endNum))
            }
          }
          else if(binding.value.item.useApiRemote){
            binding.value.item.loadQuery = { pageNo: 1, pageSize: onceLoad }
            vnode.context.$emit(binding.value.item.attrs.remoteMethod ? binding.value.item.attrs.remoteMethod : 'loadRemote',{ isFirst: true, item: binding.value.item, load: true, queryStr: false, ...binding.value.item.loadQuery })
          }
          let beforeTop = 0
          const selectDOM = el.querySelector('.el-select-dropdown .el-select-dropdown__wrap')
          selectDOM.addEventListener('scroll', function() {
            if(this.scrollTop - beforeTop >= 0) {
              beforeTop = this.scrollTop
              const result = this.scrollHeight - this.scrollTop - 3 <= this.clientHeight
              if (result) {
                if(!binding.value.item.useApiRemote) {
                  if(nowCount >= loadLen - 1) return
                  nowCount++
                  startNum = nowCount * onceLoad
                  endNum = (nowCount + 1) * onceLoad
                  if(Array.isArray(binding.value.item.childOptions)) {
                    binding.value.item.childOptions.push(...allOptions.slice(startNum, endNum))
                  }else {
                    binding.value.item.childOptions.options.push(...allOptions.slice(startNum, endNum))
                  }
                }
                else if(binding.value.item.useApiRemote){
                  binding.value.item.loadQuery.pageNo++
                  if(binding.value.item.loadQuery['totalPage'] < binding.value.item.loadQuery.pageNo) return
                  vnode.context.$emit(binding.value.item.attrs.remoteMethod ? binding.value.item.attrs.remoteMethod : 'loadRemote',{ isFirst: false, item: binding.value.item, load: true, queryStr: false, ...binding.value.item.loadQuery })
                }
              }
            }
          })
        }
      },
      update(el, binding, vnode, oldVnode) {
        if(binding.value.item.comp === 'el-select' && binding.value.item.useApiRemote && binding.value.item.attrs && !binding.value.item.attrs.popperAppendToBody) {
          const popDOM = el.querySelector('.el-select-dropdown')
          if(binding.value.item.attrs.apiloading) {
            if(!popDOM) return
            const ele = document.createElement('div')
            ele.classList = 'select-loadM'
            ele.innerHTML = `<i class="el-icon-loading"></i>`
            popDOM.appendChild(ele)
          }else {
            const ele = el.querySelector('.el-select-dropdown .select-loadM')
            if(ele) {
              ele.remove()
            }
          }
        }
      },
    }
  }
}
</script>

<template>
  <div class="searchPdL" :class="{ 'form-row': alongContentWidth }">
    <el-row :gutter="gutter">
      <el-form :model="formModel" ref="formModel" :label-position="labelPosition" :label-width="labelWidth" :disabled="formDisabled" :formAttrs="formAttrs">
        <el-col :span="item.span" v-for="item in formItems" :key="item.prop" v-if="!item.hidden" v-bind="item.colAttrs" >
          <el-form-item class="flex-label" :rules="item.required ? item.rules ? item.rules : { required: true, trigger: 'change', message: `${handleRuleMsg(item.label)}` } : null" :prop="item.prop" v-bind="item.itemAttrs">
            <slot v-if="item.slot" :name="item.slot"></slot>
            <template slot="label">
              <template v-if="!item.isBtn">
                <el-tooltip v-if="item.label && item.label.length > labelHiddenLength" :content="dealLabel(item.label)" placement="top">
                  <span class="overflow-box"> {{ item.label }} </span>
                </el-tooltip>
                <span v-else class="overflow-box"> {{ item.label }} </span>
              </template>
              <component v-else-if="item.isBtn" :is="item.comp" v-bind="item.attrs" :disabled="item.btnDisabled" @click="handleApiForm(item)" > {{ item.label }} </component>
            </template>
            <component v-if="!item.isBtn" :is="item.comp" v-model="formModel[item.prop]" :placeholder="item.placeholder ? item.placeholder : handlePlaceHoder(item)" v-lazyLoad="{ isLazy: !!item.lazy, item }" v-bind="item.attrs ? { ...defaultAttrs, ...item.attrs } : defaultAttrs" :remote-method="item.attrs && item.attrs.remoteMethod ? q => remoteMethod(q, item) : null" @focus="e => handleFocus(e, item.focus, { item, formModel })" @change="handleChange(item.change, { item, formModel })" @handleUpdateComp="handleUpdateComp">
              <template v-if="item.childOptions">
                <component :is="childItem.comp" v-for="childItem in dealChildOptions(item.childOptions, item)" :key="childItem.label" :style="{'min-width': item.comp === 'el-select' ? `${selectOpMinWid}px` : 'initial'}" v-bind="childItem.attrs">
                  <template v-if="(childItem.label || childItem.label === 0) && childItem.attrs && childItem.attrs.value">{{ childItem.attrs.value }}</template>
                </component>
                <template v-if="item.comp === 'el-select'" slot="empty">
                  <div class="empty-select-box" :style="{ width: `${selectOpMinWid}px` }">无数据</div>
                </template>
              </template>
            </component>
          </el-form-item>
        </el-col>
      </el-form>
    </el-row>
    <slot></slot>
  </div>
</template>

<script>
export default {
  name: "searchBtns",
  props: {
    actionList: {
      type: Array,
      default() {
        return []
      }
    }
  },
  computed: {
    getWidth() {
      let measureWid = 0
      this.actionList.forEach(item => {
        // letter width
        measureWid += item.label ? item.label.length * 17 : 0
      })
      // element padding
      measureWid += this.actionList.length * 37 + (this.actionList.length - 1) * 15
      return `${measureWid}px`
    }
  },
  data() {
    return {}
  },
  methods: {
    dealSrcTrans(className) {
      // 数组class 默认取第一个
      let path = ''
      if (Array.isArray(className)) {
        path = className.split(' ')[0]
      } else {
        path = className
      }

      switch (path) {
        case 'search':
          return 'search.png'
        case 'add':
          return ''
        case 'remove':
          return ''
        case 'reset':
          return 'refresh.png'
        case 'import':
          return ''
        case 'download':
          return ''
        case 'export':
          return ''
        default:
          return 'search.png'
      }
    },
    handleClick(code) {
      this.$emit('handleQueryOptions', { code })
    }
  }
}
</script>

<template>
  <div class="operation-btn" :style="{ '--width': getWidth }">
    <button v-for="item in actionList" v-if="!item.hidden" v-has-permission="item.perKey" @click="handleClick(item.code)">
      <img v-if="!item.icon" alt="" :src="item.src ? item.src : require(`@/assets/icon/search/${dealSrcTrans(item.class)}`)" />
      <i v-else :class="item.icon"></i>
      <span v-if="item.label"> {{ item.label }} </span>
    </button>
  </div>
</template>

<script>
import { cloneDeep, isArray, isObject } from "lodash-es"
import { resizeEl } from "@/components/chart/resizeEl"
let listModel = null
const optType = {
  DELETE: 'delete',
}
export default {
  name: "tableComp",
  mixins: [resizeEl],
  components: {
    renderDom: {
      functional: true,
      props: {
        render: Function,
        row: Object
      },
      render: (h, context) => {
        return context.props.render(h, context.props)
      }
    }
  },
  props: {
    tableColumns: {
      type: Array,
      default() {
        return []
      }
    },
    canSelect: {
      type: Boolean,
      default: false
    },
    showIndex: {
      type: Boolean,
      default: true
    },
    hasPagination: {
      type: Boolean,
      default: true
    },
    selectHeadAttrs: {
      type: Object,
      default() {
        return {
          width: 55
        }
      }
    },
    indexHeadAttrs: {
      type: Object,
      default() {
        return {
          width: 60
        }
      }
    },
    dataObject: {
      type: Object,
      default() {
        return {
          data: 'data',
          total: 'count'
        }
      }
    },
    useNoFetch: {
      type: Boolean,
      default: false
    },
    tableDataWithNoFetch: {
      type: Object,
      default() {
        return {
          data: [],
          total: 0
        }
      }
    }
  },
  data() {
    return {
      tableData: [],
      query: {
        pageNum: 1,
        pageSize: 20
      },
      total: 0,
      tableH: 400,
      tableLoading: false,
      selectOptions: [],
      handleOptions: {},
      copyParams: null,
      show: true
    }
  },
  methods: {
    handleSelectionChange(list) {
      this.selectOptions = cloneDeep(list)
    },
    getSelection() {
      return this.selectOptions
    },
    handleSortChange({ column, prop, order }) {
      if(!this.useNoFetch) {
        const orderBy = {
          ascending: 'asc',
          descending: 'desc'
        }
        const obj = { ...this.copyParams, sortName: prop, sortOrder: orderBy[order] }
        this.handleSearch(obj, listModel, true)
      }
    },
    handleChange(v, type) {
      if(type === 'page') {
        this.query.pageNum = v
      }else {
        this.query.pageNum = 1
        this.query.pageSize = v
      }

      if(this.useNoFetch) {
        return this.$emit('tablePageChange', this.query)
      }

      this.handleSearch(this.copyParams, listModel, false)
    },
    chooseOperation(type, row, hasDisable) {
      if(hasDisable) return
      if(type === optType.DELETE) {
        this.$confirm("确认删除？", "删除", {
          customClass: 'customConfirm',
          cancelButtonClass: 'btn-custom-cancel',
          confirmButtonText: '确定',
          cancelButtonText: '取消',
          type: "warning",
        }).then(_ => {
          this.$emit('handleOperation', { type, row })
        })
        return
      }
      this.$emit('handleOperation', { type, row })
    },
    handleSearch(params, apiModel, resetPageNo = true) {
      if(this.useNoFetch) return // 使用props传入的表格数据
      if(apiModel && typeof apiModel === 'function') {
        this.tableLoading = true
        !listModel ? listModel = apiModel : ''
        this.copyParams = params
        const modelParams = { ...(this.hasPagination ? this.query : {}), ...params }
        if(resetPageNo) {
          this.hasPagination ? modelParams['pageNum'] = 1 : ''
          this.query.pageNum = 1
        }
        apiModel(modelParams).then(res => {
          this.tableLoading = false
          if(res.code && res.code === 200) {
            if(!res.data) return // data 会返回null
            const dataPath = (result, keyName) => {
              let data = result, getPath = this.dataObject[`${keyName}`]
              if(!getPath) return console.warn(`返回数据无 ${keyName} 字段`)
              const path = this.dataObject[`${keyName}`].split('.')
              for (const pathElement of path) {
                if(data[pathElement] || data[pathElement] === 0) {
                  data = data[pathElement]
                }else {
                  data = null
                  break
                }
              }
              return data
            }
            const data = dataPath(res, 'data')
            const number = dataPath(res, 'total')
            if(isArray(data)) {
              this.tableData = cloneDeep(data)
              this.total = number && number !== 0 ? Number(number) : 0
            }
            return
          }
          this.$message.error(res.msg)
          this.tableData.splice(0)
          this.total = 0
        }).catch(_ => {
          this.tableLoading = false
        })
      }else {
        console.error('请传入列表接口！')
      }
    },
    dealHeight() {
      // 剩余表格高度
      this.tableH = this.$refs.tableComp.clientHeight
      this.$nextTick(()=>{
        this.$refs.table.$el.style.height = `${this.tableH - (this.hasPagination ? 30 : 0)}px`
      })
    },
    resetData() {
      this.tableData.splice(0)
      this.total = 0
      if(this.hasPagination) {
        this.query.pageNum = 1
      }
    }
  },
  created() {
    this.handleOptions['handleSearch'] = this.handleSearch
    this.handleOptions['handleSelect'] = this.getSelection
    this.$emit('tableActions', this.handleOptions)
  },
  computed: {
    getTableH() {
      return `${this.tableH - (this.hasPagination ? 48 : 0)}px`
    }
  },
  watch: {
    tableDataWithNoFetch: {
      handler(newVal) {
        if(this.useNoFetch && isObject(newVal) && isArray(newVal.data) && newVal.data.length) {
          this.tableData = cloneDeep(newVal.data)
          this.total = newVal.total
        }else {
          this.tableData.splice(0)
          this.total = 0
        }
      },
      deep: true
    },
    show(newVal) {
      if(newVal) {
        this.dealHeight()
      }
    }
  },
  mounted() {
    this.dealHeight()
  }
}
</script>

js
let resizeObserver  = null

export const resizeEl = {
    mounted() {
        resizeObserver = new ResizeObserver(entries => {
            this.chartIns && this.chartIns.resize()
        })

        this.$refs.equipRef && resizeObserver && resizeObserver.observe(this.$refs.equipRef)

        // F11情况下grid布局
        window.addEventListener('resize', () => {
            this.show = false
            setTimeout(() => {
                this.show = true
            })
        })
    },
    beforeDestroy() {
        this.$refs.equipRef && resizeObserver && resizeObserver.unobserve(this.$refs.equipRef)
        resizeObserver = null
    }
}

<script>
import { resizeEl } from "@/components/chart/resizeEl"
import { cloneDeep } from 'lodash-es'

export default {
  name: "chartComp",
  mixins: [resizeEl],
  props: {
    chartOptions: {
      type: Object,
      default() {
        return {}
      }
    }
  },
  data() {
    return {
      chartIns: null,
      show: true,
      zoom: null
    }
  },
  watch: {
    chartOptions: {
      handler(newVal) {
        const obj = cloneDeep(newVal)
        this.chartIns && this.chartIns.clear()
        this.chartIns && this.chartIns.setOption(obj)
      },
      deep: true,
    },
    show(newVal) {
      if(newVal) {
        this.setZoom()
        this.$nextTick(() => {
          this.chartIns = this.$echarts.init(this.$refs.chartContain)
          this.chartIns && this.chartIns.setOption(cloneDeep(this.chartOptions))
        })
      }
    },
    isThemeLight(newVal) {
      this.dealInit()
    }
  },
  methods: {
    setZoom() {
      this.zoom = document.getElementsByTagName('body')[0].style.zoom || 1
    },
    dealInit() {
      this.$nextTick(() => {
        const obj = cloneDeep(this.chartOptions)
        this.chartIns && this.chartIns.clear()
        this.chartIns && this.chartIns.setOption(obj)
      })
    }
  },
  mounted() {
    this.chartIns = this.$echarts.init(this.$refs.chartContain)
    this.setZoom()
    this.dealInit()
  }
}
</script>
